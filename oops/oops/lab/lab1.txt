\section*{4. Function to calculate quotient and remainder}

\subsection*{Code}
\begin{lstlisting}[style=cpp]
void fxn4(int a, int b, int &quotient, int &remainder) {
    if (b == 0) {
        cout << "Error: Division by zero is undefined." << endl;
        quotient = 0;
        remainder = 0;
        return;
    }
    quotient = a / b;
    remainder = a % b;
}

int main4() {
    int num1 = 20, num2 = 6;
    int q, r;
    fxn4(num1, num2, q, r);

    return 0;
}
